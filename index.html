# initialise variables
snowfall_total = 0
measurement_counter = 0
not_included = 0
stored_measurements = []
days = 0

# loop for a maximum number of entries
for i in range(100):

    snowfall = float(input("Enter snowfall (in cm): "))

    # check input is between 0 and 60
    if snowfall < 0 or snowfall > 60:
        print("Try again. Your value must be between 0 and 60.")
        continue

    # stop entering data when 0 is entered
    if snowfall == 0:
        break

    # count number of days snowfall lasted
    days += 1

    # store measurements that are 0.5 or higher
    if snowfall >= 0.5:
        stored_measurements.append(snowfall)
        snowfall_total += snowfall
        measurement_counter += 1
    else:
        # count measurements not included
        not_included += 1


# check if there is enough data for analysis
if snowfall_total >= 10 and days >= 5 and measurement_counter > 0:

    # calculate average snowfall
    average = snowfall_total / measurement_counter

    # display results
    print("The number of days snowfall lasted was: " + str(days))
    print(str(not_included) + " days were not included in the calculations")
    print("The " + str(measurement_counter) + " stored measurements were as follows:")

    for i in range(len(stored_measurements)):
        print(stored_measurements[i])

    print("The average of the stored measurements is: " + str(round(average, 2)))

# display message if conditions not met
else:
    print("Not enough data for analysis.")
